#include <LedControl.h>
LedControl lc(11,13,10,2);
byte A[8]={
B00111100,B01000010,B10000001,B10000001,
B11111111,B10000001,B10000001,B10000001};
byte B[8]={
B11111110,B10000001,B10000001,B11111110,
B10000001,B10000001,B10000001,B11111110};
void draw(int d,byte p[]){
  for(int r=0;r<8;r++)
    for(int c=0;c<8;c++)
      lc.setLed(d,r,c,bitRead(p[r],7-c));
}
void setup(){
  for(int i=0;i<2;i++){
    lc.shutdown(i,false);
    lc.set Intensity(i,8);
  }
  draw(0,A);
  draw(1,B);
}
void loop(){}
